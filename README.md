# CaptchaSolver
Growtopia captcha solver using C

# Example
`#include <stdio.h>
#include "captchaSolver.h"

int main() {
	printf("%s\n", captchaSolving("http://ubistatic-a.akamaihd.net/0098/captcha/generated/8c2dbe94-5dfe-4508-93d6-fb12d08d31c7-PuzzleWithMissingPiece.rttex"));
	return 0;
}`
